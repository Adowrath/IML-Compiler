Parallel Assignment


// with var a: int64; var b: int64; var c: int64;
a init, b init := 1, 2;
// a = 1, b = 2
a, b, c init := b, a, a + b;
// a = 2, b = 1, c = 3

Ternäres Konditional


// with var a: int64; var b: int64; var c: int64;
a init := 12; b init := 20;

c init := a > b ? 1 : a < b ? -1 : 0;
// c = -1

proc divide(in  copy const m:int64, in  copy const n:int64,
            out ref  var   q:int64, out ref  var   r:int64)
do
  q init, r init := 0, m;
  while r >= n do
    q, r := q + 1, r - n
  endwhile
endproc;
