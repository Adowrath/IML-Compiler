Parallel Assignment

Vorbild Python 

							a, b, c = 1, 2, 3
							print(a, b, c) // "1 2 3"
IML 

								// with var a, b, c: int64;
								a init, b init := 1, 2;
								// a = 1, b = 2
								a, b, c init := b, a, a + b;
								// a = 2, b = 1, c = 3

Beispiele

Parallel Assignment

Fibonacci


							fun fib(var n: int64)
							  returns m: int64
							local
							  t: int64
							do
							  m init, t init := 0, 1;
							  while n > 0 do
							    m, t, n = t, m + t, n - 1
							  endwhile
							endfun

Iteratives Faktorial


							fun fact(n: int64)
							  returns acc: int64
							local
							  i: int64
							do
							  acc init, i init := 1, 1;
							  while i <= n do
							    acc, i := acc * i, i + 1
							  endwhile
							endfun

Iteratives Faktorial


							proc divide(in copy const m:int64, in copy const n:int64,
							            out ref var q:int64, out ref var r:int64)
							do
							  q init, r init := 0, m;
							  while r >= n do
							    q, r := q + 1, r - n
							  endwhile
							endproc

Ternäres Konditional

Max-Funktion


							fun max(a: int64, b: int64)
							  returns m: int64
							do
							  m init := a > b ? a : b
							endfun

Rekursives Faktorial


							fun fact(n: int64)
							  returns m: int64
							do
							  m init := n <= 0 ? 1 : n * fact(n - 1)
